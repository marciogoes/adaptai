

@router.post("/gerar-pei-de-relatorios")
async def gerar_pei_de_relatorios(
    data: dict,
    current_user: User = Depends(get_current_active_user),
    db: Session = Depends(get_db)
):
    """üéØ GERA PEI COMPLETO AUTOMATICAMENTE A PARTIR DOS RELAT√ìRIOS SALVOS"""
    student_id = data.get("student_id")
    relatorio_ids = data.get("relatorio_ids", [])
    
    if not student_id:
        raise HTTPException(status_code=400, detail="student_id √© obrigat√≥rio")
    
    if not relatorio_ids or len(relatorio_ids) == 0:
        raise HTTPException(status_code=400, detail="Selecione pelo menos um relat√≥rio")
    
    # Verificar se aluno existe
    student = db.query(Student).filter(Student.id == student_id).first()
    if not student:
        raise HTTPException(status_code=404, detail="Aluno n√£o encontrado")
    
    # Carregar relat√≥rios
    relatorios = db.query(Relatorio).filter(
        Relatorio.id.in_(relatorio_ids),
        Relatorio.student_id == student_id
    ).all()
    
    if len(relatorios) == 0:
        raise HTTPException(status_code=404, detail="Nenhum relat√≥rio encontrado")
    
    # Compilar dados de todos os relat√≥rios
    relatorios_dados = []
    for rel in relatorios:
        dados = rel.dados_extraidos
        
        # Carregar JSON completo se existir
        if isinstance(dados, dict) and dados.get("json_path"):
            json_file = RELATORIOS_DIR / dados["json_path"]
            if json_file.exists():
                try:
                    with open(json_file, 'r', encoding='utf-8') as f:
                        dados = json.load(f)
                except:
                    pass
        
        relatorios_dados.append({
            "id": rel.id,
            "tipo": rel.tipo,
            "profissional": {
                "nome": rel.profissional_nome,
                "especialidade": rel.profissional_especialidade,
                "registro": rel.profissional_registro
            },
            "data_emissao": rel.data_emissao.isoformat() if rel.data_emissao else None,
            "resumo": rel.resumo,
            "dados_completos": dados
        })
    
    # Chamar IA para gerar PEI
    client = get_anthropic_client()
    if not client:
        raise HTTPException(status_code=500, detail="Servi√ßo de IA n√£o dispon√≠vel")
    
    prompt = f"""Voc√™ √© um especialista em educa√ß√£o inclusiva criando um PEI completo.

INFORMA√á√ïES DO ALUNO:
- Nome: {student.name}
- S√©rie/Ano: {student.grade_level or 'N√£o especificado'}

RELAT√ìRIOS DE TERAPIAS E ACOMPANHAMENTO ({len(relatorios_dados)} documentos):
{json.dumps(relatorios_dados, ensure_ascii=False, indent=2)}

Com base em TODOS os relat√≥rios acima, gere um PEI COMPLETO e DETALHADO em formato JSON:

{{
    "diagnosticos": {{
        "tea": false,
        "tea_nivel": null,
        "tdah": false,
        "dislexia": false,
        "discalculia": false,
        "disgrafia": false,
        "deficiencia_visual": false,
        "deficiencia_auditiva": false,
        "deficiencia_intelectual": false,
        "deficiencia_fisica": false,
        "superdotacao": false,
        "outro": false,
        "outro_qual": ""
    }},
    "caracteristicas_gerais": "Par√°grafo detalhado (3-4 linhas)",
    "pontos_fortes": "Par√°grafo detalhado",
    "dificuldades": "Par√°grafo detalhado",
    "adaptacoes_curriculares": "Par√°grafo detalhado",
    "adaptacoes_avaliacao": "Par√°grafo detalhado",
    "adaptacoes_ambiente": "Par√°grafo detalhado",
    "recursos_apoio": "Par√°grafo detalhado",
    "metas_curto_prazo": "Par√°grafo com 3-5 metas para 1-3 meses",
    "metas_medio_prazo": "Par√°grafo com 3-5 metas para 3-6 meses",
    "metas_longo_prazo": "Par√°grafo com 3-5 metas para o ano",
    "estrategias_ensino": "Par√°grafo detalhado",
    "estrategias_comunicacao": "Par√°grafo detalhado",
    "estrategias_comportamento": "Par√°grafo detalhado",
    "profissionais_apoio": "Lista de profissionais",
    "frequencia_acompanhamento": "Frequ√™ncia de revis√£o",
    "observacoes": "Observa√ß√µes importantes"
}}

INSTRU√á√ïES:
- Analise TODOS os relat√≥rios e consolide as informa√ß√µes
- Seja ESPEC√çFICO e PR√ÅTICO - evite generalidades
- Texto corrido (sem bullets)
- M√≠nimo 3-4 linhas por campo
- Marque diagn√≥sticos apenas se mencionados
- Seja encorajador mas realista

Retorne APENAS o JSON."""

    try:
        print(f"ü§ñ Gerando PEI completo para {student.name} com {len(relatorios_dados)} relat√≥rios...")
        
        message = client.messages.create(
            model="claude-sonnet-4-20250514",
            max_tokens=8000,
            messages=[
                {
                    "role": "user",
                    "content": prompt
                }
            ],
        )
        
        response_text = message.content[0].text.strip()
        
        # Limpar marcadores de c√≥digo
        for marker in ["```json", "```"]:
            response_text = response_text.replace(marker, "")
        response_text = response_text.strip()
        
        try:
            pei_gerado = json.loads(response_text)
        except json.JSONDecodeError as e:
            print(f"‚ùå Erro ao parsear JSON: {e}")
            pei_gerado = {
                "erro_parse": True,
                "texto_bruto": response_text,
                "mensagem": "N√£o foi poss√≠vel estruturar o PEI automaticamente"
            }
        
        print(f"‚úÖ PEI gerado com sucesso para {student.name}!")
        
        return {
            "success": True,
            "student_name": student.name,
            "relatorios_utilizados": len(relatorios_dados),
            "pei": pei_gerado
        }
        
    except Exception as e:
        print(f"‚ùå Erro ao gerar PEI: {e}")
        raise HTTPException(
            status_code=500,
            detail=f"Erro ao gerar PEI com IA: {str(e)}"
        )
